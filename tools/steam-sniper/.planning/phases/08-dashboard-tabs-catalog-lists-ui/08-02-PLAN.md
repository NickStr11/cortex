   ```javascript
   const [favRes, wishRes] = await Promise.all([
     fetch(`/api/lists?user=${USER}&type=favorite`),
     fetch(`/api/lists?user=${USER}&type=wishlist`),
   ]);

   const wishData = await wishRes.json();
   ```
2. Rebuild sets: `_favorites = new Set(favData.items.map(i => i.item_name))`

**`async function toggleListItem(itemName, listType)`** -- internal:
1. Check if item is currently in the list: `const inList = isInList(itemName, listType)`
2. If in list, DELETE:
   ```javascript
   await fetch('/api/lists', {
     method: 'DELETE',
     headers: { 'Content-Type': 'application/json' },
     body: JSON.stringify({ user: USER, item_name: itemName, list_type: listType }),
   });
   ```
   Remove from set: `_favorites.delete(itemName)` or `_wishlist.delete(itemName)`
3. If not in list, POST:
   ```javascript
   await fetch('/api/lists', {
     method: 'POST',
     headers: { 'Content-Type': 'application/json' },
     body: JSON.stringify({ user: USER, item_name: itemName, list_type: listType }),
   });
   ```
   Add to set: `_favorites.add(itemName)` or `_wishlist.add(itemName)`
4. Update the toggle button visually immediately (optimistic UI):
   - Find all buttons matching `[data-list="${listType}"][data-name="${CSS.escape(itemName)}"]` (multiple may exist in catalog + list tab)
   - Toggle class: `active-fav` for favorite, `active-wish` for wishlist
   - Toggle innerHTML: `&#9829;` (filled) / `&#9825;` (empty) for heart, `&#9733;` / `&#9734;` for star
5. Emit `events.emit('lists:changed', { itemName, listType, action: inList ? 'removed' : 'added' })`

Build cards for list tab:
```javascript
function renderListTab(listType, containerId) {
  const container = document.getElementById(containerId);
  const items = listType === 'favorite' ? _favorites : _wishlist;

  if (items.size === 0) {
    const msg = listType === 'favorite'
      ? 'No favorites yet -- click \u2661 on catalog cards'
      : 'Wishlist is empty -- click \u2606 on catalog cards';
    container.innerHTML = `<div class="empty-state">${msg}</div>`;
    return;
  }

  let html = `<div class="list-tab-header"><span class="section-title">${listType === 'favorite' ? 'Favorites' : 'Wishlist'} <span class="badge">${items.size}</span></span></div>`;
  html += '<div class="list-tab-grid">';

  for (const name of items) {
    const cached = _priceCache.get(name);
    const imgSrc = cached && cached.url
      ? 'https://community.akamai.steamstatic.com/economy/image/' + cached.url + '/200fx200f'
      : '';
    const imgHtml = imgSrc
      ? `<img class="cat-card-img" src="${imgSrc}" loading="lazy" alt="">`
      : '<div class="cat-card-img-empty"></div>';
    const price = cached ? fmtRub(cached.price_rub) : '--';
    const catBadge = cached ? cached.category : '';
    const count = cached ? cached.count : '?';
    const safeName = name.replace(/"/g, '&quot;');

    html += `<div class="cat-card" data-item-name="${safeName}">
      <div class="cat-card-actions">
        <button class="list-toggle ${listType === 'favorite' ? 'active-fav' : 'active-wish'}"
          data-list="${listType}" data-name="${safeName}"
          title="Remove">${listType === 'favorite' ? '\u2665' : '\u2605'}</button>
      </div>
      ${imgHtml}
      <div class="cat-card-name">${name}</div>
      <div class="cat-card-bottom">
        <div class="cat-card-price">${price}</div>
        <div class="cat-card-meta">
          ${catBadge ? `<div class="cat-card-category">${catBadge}</div>` : ''}
          <div class="cat-card-count">${count} qty</div>
        </div>
      </div>
    </div>`;
  }

  html += '</div>';
  container.innerHTML = html;
}
```
